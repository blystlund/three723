async function callClaude(system,user){

  const r=await fetch('https://api.anthropic.com/v1/messages',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({model:'claude-sonnet-4-20250514',max_tokens:1000,system,messages:[{role:'user',content:user}]})});

// ── Voice ─────────────────────────────────────────────────────────────

function toggleRecording(){

  const SR=window.SpeechRecognition||window.webkitSpeechRecognition;

  if(!SR){alert('Speech recognition not supported. Please type.');return}

  if(S.recording){S.recognition?.stop();S.recording=false;render();return}

  const rec=new SR();rec.continuous=true;rec.interimResults=true;rec.lang='en-US';

  let final='';

  rec.onresult=e=>{let interim='';for(let i=e.resultIndex;i<e.results.length;i++){if(e.results[i].isFinal)final+=e.results[i][0].transcript+' ';else interim+=e.results[i][0].transcript;}S.journalText=final+interim;const ta=document.getElementById('journal-ta');if(ta)ta.value=S.journalText;};

  rec.onend=()=>{S.recording=false;render()};

  rec.start();S.recognition=rec;S.recording=true;render();

}

function toggleBibleEntry(key,label){

function toggleTask(id){

  updateProject(p=>{

    const task=p.tasks.find(t=>t.id===id);

    const newDone=!task?.done;

    if(newDone)addLog('task','Completed: '+task?.text,'var(--green)');

    return{...p,tasks:p.tasks.map(t=>t.id===id?{...t,done:!t.done}:t)};

  });

  render();

}


function taskDistraction(taskId){

  S.activeTaskId=taskId;

  const note='Task distraction';

  updateProject(p=>({...p,distractions:[...(p.distractions||[]),{note,time:new Date().toLocaleTimeString('en-US',{hour:'numeric',minute:'2-digit'}),taskId}]}));

  addLog('distraction','⚡ '+note,'var(--red)');

  S.timerSecs=600;S.timerRunning=true;S.timerDone=false;S.distractionNote=note;

  S.view='distractionTimer';render();

  clearInterval(timerInterval);

  timerInterval=setInterval(()=>{

    S.timerSecs--;

    if(S.timerSecs<=0){S.timerSecs=0;S.timerRunning=false;S.timerDone=true;clearInterval(timerInterval);}

    renderTimerLive();

  },1000);

}


// ── DISTRACTION TIMER ─────────────────────────────────────────────────

function renderDistractionTimer(){

  const p=getProject();

  const task=p?.tasks.find(t=>t.id===S.activeTaskId);

  const R=96,circ=2*Math.PI*R;

  const pct=(600-S.timerSecs)/600;

  const col=S.timerDone?'var(--green)':'var(--red)';

  const isFlashing=S.timerRunning&&!S.timerDone;

  return`<div class="fade-in">

  <div class="back-row"><button class="back-btn" onclick="clearInterval(timerInterval);S.timerRunning=false;S.view=S.activeProjectId?'projectDetail':'home';render()">‹</button><span class="back-title">Distraction Mode</span></div>

  <div style="padding:0 20px 20px">

    ${task?`<div style="background:var(--surface2);border-radius:12px;padding:12px 16px;margin-bottom:20px;font-size:14px;color:var(--text-sub)">Pausing: <span style="color:var(--text);font-weight:600">${task.text}</span></div>`:''}

    <div style="background:var(--surface);border-radius:20px;border:0.5px solid ${isFlashing?'var(--red)':'var(--border)'};padding:30px 20px;margin-bottom:16px;${isFlashing?'animation:borderFlash 0.8s ease-in-out infinite':''}" id="timer-card">

      <div class="timer-wrap">

        <svg class="timer-svg" width="210" height="210">

          <circle cx="105" cy="105" r="${R}" fill="none" stroke="var(--surface2)" stroke-width="10"/>

          <circle id="timer-fill" cx="105" cy="105" r="${R}" fill="none" stroke="${col}" stroke-width="10"

            stroke-dasharray="${circ}" stroke-dashoffset="${circ-circ*pct}" stroke-linecap="round"

            style="transition:stroke-dashoffset 1s linear${isFlashing?';animation:timerFlash 0.6s ease-in-out infinite':''}"/>

        </svg>

        <div class="timer-center">

          <div id="timer-number" class="timer-num" style="color:${col}">${fmt(S.timerSecs)}</div>

          <div id="timer-label" class="timer-lbl">${S.timerDone?"Time's up!":'distraction time'}</div>

        </div>

      </div>

    </div>

    <input id="distraction-note" class="text-input" placeholder="What distracted you?" value="${S.distractionNote}" style="margin-bottom:16px" oninput="S.distractionNote=this.value">

    ${!S.timerDone?`<div style="display:flex;gap:10px">

      <button class="btn-record ${S.timerRunning?'recording':''}" onclick="${S.timerRunning?'pauseTimerBtn()':'resumeTimerBtn()'}" style="flex:1">

        <span>${S.timerRunning?'⏸ Pause':'▶ Resume'}</span>

      </button>

      <button onclick="finishDistraction()" style="flex:1;padding:16px;border-radius:16px;background:var(--green);border:none;color:#000;font-size:15px;font-weight:700;cursor:pointer;font-family:inherit">Back ✓</button>

    </div>`:

    `<button onclick="finishDistraction()" class="btn-primary" style="background:var(--green);color:#000">✓ Back to Work</button>

    <div style="text-align:center;margin-top:18px;font-size:15px;font-style:italic;color:var(--text-sub)">Refocus and finish strong.</div>`}

  </div></div>`;

}

function pauseTimerBtn(){S.timerRunning=false;clearInterval(timerInterval);render()}

function resumeTimerBtn(){

  S.timerRunning=true;

  timerInterval=setInterval(()=>{

    S.timerSecs--;

    if(S.timerSecs<=0){S.timerSecs=0;S.timerRunning=false;S.timerDone=true;clearInterval(timerInterval);}

    renderTimerLive();

  },1000);

  render();

}

function finishDistraction(){

  clearInterval(timerInterval);S.timerRunning=false;

  S.view=S.activeProjectId?'projectDetail':'home';render();

}
